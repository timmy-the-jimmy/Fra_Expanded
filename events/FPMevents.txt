#Orleanist death event
country_event = {
	id = fpmleagues.11
	title = fpmleagues.11.t
	desc = fpmleagues.11.d
	picture = GFX_report_event_generic_funeral

	trigger = {
		has_completed_focus = FRA_orleanist_restoration
		date > 1940.8.25
		OR = {
			has_country_leader = {
				character = FRA_jean_duke_of_guise
			}
			has_idea = FPM_jean_3
		}
	}
	mean_time_to_happen = {
		days = 365
	}
	fire_only_once = yes

	option = {
		name = fpmleagues.11.a
		if = {
			limit = {
				OR = {
					has_completed_focus = FRA_action_francais
					has_completed_focus = FRA_constitutional_monarchy
				}
			}
			swap_ideas = {
				remove_idea = FPM_jean_3
				add_idea = FPM_henri_6
			}
		}
		else = {
			kill_country_leader = yes
			add_country_leader_role = {
				character = FRA_henry_count_of_paris
				country_leader = {
					expire = "1965.1.1"
					ideology = moderatism
					traits = {
						count_of_paris
					}
				}
				promote_leader = yes
			}
		}
	}
}
